23class Config(object): 

24 """Advanced configuration for Botocore clients. 

25 

26 :type region_name: str 

27 :param region_name: The region to use in instantiating the client 

28 

29 :type signature_version: str 

30 :param signature_version: The signature version when signing requests. 

31 

32 :type user_agent: str 

33 :param user_agent: The value to use in the User-Agent header. 

34 

35 :type user_agent_extra: str 

36 :param user_agent_extra: The value to append to the current User-Agent 

37 header value. 

38 

39 :type connect_timeout: float or int 

40 :param connect_timeout: The time in seconds till a timeout exception is 

41 thrown when attempting to make a connection. The default is 60 

42 seconds. 

43 

44 :type read_timeout: float or int 

45 :param read_timeout: The time in seconds till a timeout exception is 

46 thrown when attempting to read from a connection. The default is 

47 60 seconds. 

48 

49 :type parameter_validation: bool 

50 :param parameter_validation: Whether parameter validation should occur 

51 when serializing requests. The default is True. You can disable 

52 parameter validation for performance reasons. Otherwise, it's 

53 recommended to leave parameter validation enabled. 

54 

55 :type max_pool_connections: int 

56 :param max_pool_connections: The maximum number of connections to 

57 keep in a connection pool. If this value is not set, the default 

58 value of 10 is used. 

59 

60 :type proxies: dict 

61 :param proxies: A dictionary of proxy servers to use by protocol or 

62 endpoint, e.g.: 

63 {'http': 'foo.bar:3128', 'http://hostname': 'foo.bar:4012'}. 

64 The proxies are used on each request. 

65 

66 :type proxies_config: dict 

67 :param proxies_config: A dictionary of additional proxy configurations. 

68 Valid keys are: 

69 

70 * 'proxy_ca_bundle' -- The path to a custom certificate bundle to use 

71 when establishing SSL/TLS connections with proxy. 

72 

73 * 'proxy_client_cert' -- The path to a certificate for proxy 

74 TLS client authentication. 

75 

76 When a str is provided it is treated as a path to a proxy client 

77 certificate. When a two element tuple is provided, it will be 

78 interpreted as the path to the client certificate, and the path 

79 to the certificate key. 

80 

81 * 'proxy_use_forwarding_for_https' -- For HTTPS proxies, 

82 forward your requests to HTTPS destinations with an absolute 

83 URI. We strongly recommend you only use this option with 

84 trusted or corporate proxies. Value must be boolean. 

85 

86 :type s3: dict 

87 :param s3: A dictionary of s3 specific configurations. 

88 Valid keys are: 

89 

90 * 'use_accelerate_endpoint' -- Refers to whether to use the S3 

91 Accelerate endpoint. The value must be a boolean. If True, the 

92 client will use the S3 Accelerate endpoint. If the S3 Accelerate 

93 endpoint is being used then the addressing style will always 

94 be virtual. 

95 

96 * 'payload_signing_enabled' -- Refers to whether or not to SHA256 

97 sign sigv4 payloads. By default, this is disabled for streaming 

98 uploads (UploadPart and PutObject). 

99 

100 * 'addressing_style' -- Refers to the style in which to address 

101 s3 endpoints. Values must be a string that equals: 

102 

103 * auto -- Addressing style is chosen for user. Depending 

104 on the configuration of client, the endpoint may be addressed in 

105 the virtual or the path style. Note that this is the default 

106 behavior if no style is specified. 

107 

108 * virtual -- Addressing style is always virtual. The name of the 

109 bucket must be DNS compatible or an exception will be thrown. 

110 Endpoints will be addressed as such: mybucket.s3.amazonaws.com 

111 

112 * path -- Addressing style is always by path. Endpoints will be 

113 addressed as such: s3.amazonaws.com/mybucket 

114 

115 * 'us_east_1_regional_endpoint' - Refers to what S3 endpoint to use 

116 when the region is configured to be us-east-1. Values must be a 

117 string that equals: 

118 

119 * regional -- Use the us-east-1.amazonaws.com endpoint if the 

120 client is configured to use the us-east-1 region. 

121 

122 * legacy -- Use the s3.amazonaws.com endpoint if the client is 

123 configured to use the us-east-1 region. This is the default if 

124 the configuration option is not specified. 

125 

126 

127 :type retries: dict 

128 :param retries: A dictionary for retry specific configurations. 

129 Valid keys are: 

130 

131 * 'total_max_attempts' -- An integer representing the maximum number of 

132 total attempts that will be made on a single request. This includes 

133 the initial request, so a value of 1 indicates that no requests 

134 will be retried. If ``total_max_attempts`` and ``max_attempts`` 

135 are both provided, ``total_max_attempts`` takes precedence. 

136 ``total_max_attempts`` is preferred over ``max_attempts`` because 

137 it maps to the ``AWS_MAX_ATTEMPTS`` environment variable and 

138 the ``max_attempts`` config file value. 

139 * 'max_attempts' -- An integer representing the maximum number of 

140 retry attempts that will be made on a single request. For 

141 example, setting this value to 2 will result in the request 

142 being retried at most two times after the initial request. Setting 

143 this value to 0 will result in no retries ever being attempted on 

144 the initial request. If not provided, the number of retries will 

145 default to whatever is modeled, which is typically four retries. 

146 * 'mode' -- A string representing the type of retry mode botocore 

147 should use. Valid values are: 

148 * ``legacy`` - The pre-existing retry behavior. 

149 * ``standard`` - The standardized set of retry rules. This 

150 will also default to 3 max attempts unless overridden. 

151 * ``adaptive`` - Retries with additional client side throttling. 

152 

153 :type client_cert: str, (str, str) 

154 :param client_cert: The path to a certificate for TLS client authentication. 

155 

156 When a str is provided it is treated as a path to a client certificate 

157 to be used when creating a TLS connection. 

158 

159 If a client key is to be provided alongside the client certificate the 

160 client_cert should be set to a tuple of length two where the first 

161 element is the path to the client certificate and the second element is 

162 the path to the certificate key. 

163 

164 :type inject_host_prefix: bool 

165 :param inject_host_prefix: Whether host prefix injection should occur. 

166 

167 Defaults to True. 

168 

169 Setting this to False disables the injection of operation parameters 

170 into the prefix of the hostname. This is useful for clients providing 

171 custom endpoints that should not have their host prefix modified. 

172 """

191 def __init__(self, *args, **kwargs): 

192 self._user_provided_options = self._record_user_provided_options( 

193 args, kwargs) 

194 

195 # Merge the user_provided options onto the default options 

196 config_vars = copy.copy(self.OPTION_DEFAULTS) 

197 config_vars.update(self._user_provided_options) 

198 

199 # Set the attributes based on the config_vars 

200 for key, value in config_vars.items(): 

201 setattr(self, key, value) 

202 

203 # Validate the s3 options 

204 self._validate_s3_configuration(self.s3) 

205 

206 self._validate_retry_configuration(self.retries)

208 def _record_user_provided_options(self, args, kwargs): 

209 option_order = list(self.OPTION_DEFAULTS) 

210 user_provided_options = {} 

211 

212 # Iterate through the kwargs passed through to the constructor and 

213 # map valid keys to the dictionary 

214 for key, value in kwargs.items(): 

215 if key in self.OPTION_DEFAULTS: 

216 user_provided_options[key] = value 

217 # The key must exist in the available options 

218 else: 

219 raise TypeError( 

220 'Got unexpected keyword argument \'%s\'' % key) 

221 

222 # The number of args should not be longer than the allowed 

223 # options 

224 if len(args) > len(option_order): 

225 raise TypeError( 

226 'Takes at most %s arguments (%s given)' % ( 

227 len(option_order), len(args))) 

228 

229 # Iterate through the args passed through to the constructor and map 

230 # them to appropriate keys. 

231 for i, arg in enumerate(args): 

232 # If it a kwarg was specified for the arg, then error out 

233 if option_order[i] in user_provided_options: 

234 raise TypeError( 

235 'Got multiple values for keyword argument \'%s\'' % ( 

236 option_order[i])) 

237 user_provided_options[option_order[i]] = arg 

238 

239 return user_provided_options

241 def _validate_s3_configuration(self, s3): 

242 if s3 is not None: 

243 addressing_style = s3.get('addressing_style') 

244 if addressing_style not in ['virtual', 'auto', 'path', None]: 

245 raise InvalidS3AddressingStyleError( 

246 s3_addressing_style=addressing_style) 

247 

248 def _validate_retry_configuration(self, retries): 

249 if retries is not None: 

250 for key, value in retries.items(): 

251 if key not in ['max_attempts', 'mode', 'total_max_attempts']: 

252 raise InvalidRetryConfigurationError( 

253 retry_config_option=key) 

254 if key == 'max_attempts' and value < 0: 

255 raise InvalidMaxRetryAttemptsError( 

256 provided_max_attempts=value, 

257 min_value=0, 

258 ) 

259 if key == 'total_max_attempts' and value < 1: 

260 raise InvalidMaxRetryAttemptsError( 

261 provided_max_attempts=value, 

262 min_value=1, 

263 ) 

264 if key == 'mode' and value not in ['legacy', 'standard', 

265 'adaptive']: 

266 raise InvalidRetryModeError( 

267 provided_retry_mode=value 

268 )

270 def merge(self, other_config): 

271 """Merges the config object with another config object 

272 

273 This will merge in all non-default values from the provided config 

274 and return a new config object 

275 

276 :type other_config: botocore.config.Config 

277 :param other config: Another config object to merge with. The values 

278 in the provided config object will take precedence in the merging 

279 

280 :returns: A config object built from the merged values of both 

281 config objects. 

282 """ 

283 # Make a copy of the current attributes in the config object. 

284 config_options = copy.copy(self._user_provided_options) 

285 

286 # Merge in the user provided options from the other config 

287 config_options.update(other_config._user_provided_options) 

288 

289 # Return a new config object with the merged properties. 

290 return Config(**config_options)
