112 def unquote_str(value, encoding='utf-8'): 

113 # In python2, unquote() gives us a string back that has the urldecoded 

114 # bits, but not the unicode parts. We need to decode this manually. 

115 # unquote has special logic in which if it receives a unicode object it 

116 # will decode it to latin1. This is hard coded. To avoid this, we'll 

117 # encode the string with the passed in encoding before trying to 

118 # unquote it. 

119 byte_string = value.encode(encoding) 

120 return unquote_plus(byte_string).decode(encoding)

243def _windows_shell_split(s): 

244 """Splits up a windows command as the built-in command parser would. 

245 

246 Windows has potentially bizarre rules depending on where you look. When 

247 spawning a process via the Windows C runtime (which is what python does 

248 when you call popen) the rules are as follows: 

249 

250 https://docs.microsoft.com/en-us/cpp/cpp/parsing-cpp-command-line-arguments 

251 

252 To summarize: 

253 

254 * Only space and tab are valid delimiters 

255 * Double quotes are the only valid quotes 

256 * Backslash is interpreted literally unless it is part of a chain that 

257 leads up to a double quote. Then the backslashes escape the backslashes, 

258 and if there is an odd number the final backslash escapes the quote. 

259 

260 :param s: The command string to split up into parts. 

261 :return: A list of command components. 

262 """ 

263 if not s: 

264 return [] 

265 

266 components = [] 

267 buff = [] 

268 is_quoted = False 

269 num_backslashes = 0 

270 for character in s: 

271 if character == '\\': 

272 # We can't simply append backslashes because we don't know if 

273 # they are being used as escape characters or not. Instead we 

274 # keep track of how many we've encountered and handle them when 

275 # we encounter a different character. 

276 num_backslashes += 1 

277 elif character == '"': 

278 if num_backslashes > 0: 

279 # The backslashes are in a chain leading up to a double 

280 # quote, so they are escaping each other. 

281 buff.append('\\' * int(floor(num_backslashes / 2))) 

282 remainder = num_backslashes % 2 

283 num_backslashes = 0 

284 if remainder == 1: 

285 # The number of backslashes is uneven, so they are also 

286 # escaping the double quote, so it needs to be added to 

287 # the current component buffer. 

288 buff.append('"') 

289 continue 

290 

291 # We've encountered a double quote that is not escaped, 

292 # so we toggle is_quoted. 

293 is_quoted = not is_quoted 

294 

295 # If there are quotes, then we may want an empty string. To be 

296 # safe, we add an empty string to the buffer so that we make 

297 # sure it sticks around if there's nothing else between quotes. 

298 # If there is other stuff between quotes, the empty string will 

299 # disappear during the joining process. 

300 buff.append('') 

301 elif character in [' ', '\t'] and not is_quoted: 

302 # Since the backslashes aren't leading up to a quote, we put in 

303 # the exact number of backslashes. 

304 if num_backslashes > 0: 

305 buff.append('\\' * num_backslashes) 

306 num_backslashes = 0 

307 

308 # Excess whitespace is ignored, so only add the components list 

309 # if there is anything in the buffer. 

310 if buff: 

311 components.append(''.join(buff)) 

312 buff = [] 

313 else: 

314 # Since the backslashes aren't leading up to a quote, we put in 

315 # the exact number of backslashes. 

316 if num_backslashes > 0: 

317 buff.append('\\' * num_backslashes) 

318 num_backslashes = 0 

319 buff.append(character) 

320 

321 # Quotes must be terminated. 

322 if is_quoted: 

323 raise ValueError('No closing quotation in string: %s' % s) 

324 

325 # There may be some leftover backslashes, so we need to add them in. 

326 # There's no quote so we add the exact number. 

327 if num_backslashes > 0: 

328 buff.append('\\' * num_backslashes) 

329 

330 # Add the final component in if there is anything in the buffer. 

331 if buff: 

332 components.append(''.join(buff)) 

333 

334 return components
