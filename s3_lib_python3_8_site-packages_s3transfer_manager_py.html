478 def _submit_transfer(self, call_args, submission_task_cls, 

479 extra_main_kwargs=None): 

480 if not extra_main_kwargs: 

481 extra_main_kwargs = {} 

482 

483 # Create a TransferFuture to return back to the user 

484 transfer_future, components = self._get_future_with_components( 

485 call_args) 

486 

487 # Add any provided done callbacks to the created transfer future 

488 # to be invoked on the transfer future being complete. 

489 for callback in get_callbacks(transfer_future, 'done'): 

490 components['coordinator'].add_done_callback(callback) 

491 

492 # Get the main kwargs needed to instantiate the submission task 

493 main_kwargs = self._get_submission_task_main_kwargs( 

494 transfer_future, extra_main_kwargs) 

495 

496 # Submit a SubmissionTask that will submit all of the necessary 

497 # tasks needed to complete the S3 transfer. 

498 self._submission_executor.submit( 

499 submission_task_cls( 

500 transfer_coordinator=components['coordinator'], 

501 main_kwargs=main_kwargs 

502 ) 

503 ) 

504 

505 # Increment the unique id counter for future transfer requests 

506 self._id_counter += 1 

507 

508 return transfer_future

510 def _get_future_with_components(self, call_args): 

511 transfer_id = self._id_counter 

512 # Creates a new transfer future along with its components 

513 transfer_coordinator = TransferCoordinator(transfer_id=transfer_id) 

514 # Track the transfer coordinator for transfers to manage. 

515 self._coordinator_controller.add_transfer_coordinator( 

516 transfer_coordinator) 

517 # Also make sure that the transfer coordinator is removed once 

518 # the transfer completes so it does not stick around in memory. 

519 transfer_coordinator.add_done_callback( 

520 self._coordinator_controller.remove_transfer_coordinator, 

521 transfer_coordinator) 

522 components = { 

523 'meta': TransferMeta(call_args, transfer_id=transfer_id), 

524 'coordinator': transfer_coordinator 

525 } 

526 transfer_future = TransferFuture(**components) 

527 return transfer_future, components

554 def __exit__(self, exc_type, exc_value, *args): 

555 cancel = False 

556 cancel_msg = '' 

557 cancel_exc_type = FatalError 

558 # If a exception was raised in the context handler, signal to cancel 

559 # all of the inprogress futures in the shutdown. 

560 if exc_type: 

561 cancel = True 

562 cancel_msg = six.text_type(exc_value) 

563 if not cancel_msg: 

564 cancel_msg = repr(exc_value) 

565 # If it was a KeyboardInterrupt, the cancellation was initiated 

566 # by the user. 

567 if isinstance(exc_value, KeyboardInterrupt): 

568 cancel_exc_type = CancelledError 

569 self._shutdown(cancel, cancel_msg, cancel_exc_type)

571 def shutdown(self, cancel=False, cancel_msg=''): 

572 """Shutdown the TransferManager 

573 

574 It will wait till all transfers complete before it completely shuts 

575 down. 

576 

577 :type cancel: boolean 

578 :param cancel: If True, calls TransferFuture.cancel() for 

579 all in-progress in transfers. This is useful if you want the 

580 shutdown to happen quicker. 

581 

582 :type cancel_msg: str 

583 :param cancel_msg: The message to specify if canceling all in-progress 

584 transfers. 

585 """ 

586 self._shutdown(cancel, cancel, cancel_msg) 

587 

588 def _shutdown(self, cancel, cancel_msg, exc_type=CancelledError): 

589 if cancel: 

590 # Cancel all in-flight transfers if requested, before waiting 

591 # for them to complete. 

592 self._coordinator_controller.cancel(cancel_msg, exc_type) 

593 try: 

594 # Wait until there are no more in-progress transfers. This is 

595 # wrapped in a try statement because this can be interrupted 

596 # with a KeyboardInterrupt that needs to be caught. 

597 self._coordinator_controller.wait() 

598 except KeyboardInterrupt: 

599 # If not errors were raised in the try block, the cancel should 

600 # have no coordinators it needs to run cancel on. If there was 

601 # an error raised in the try statement we want to cancel all of 

602 # the inflight transfers before shutting down to speed that 

603 # process up. 

604 self._coordinator_controller.cancel('KeyboardInterrupt()') 

605 raise 

606 finally: 

607 # Shutdown all of the executors. 

608 self._submission_executor.shutdown() 

609 self._request_executor.shutdown() 

610 self._io_executor.shutdown()

613class TransferCoordinatorController(object): 

614 def __init__(self): 

615 """Abstraction to control all transfer coordinators 

616 

617 This abstraction allows the manager to wait for inprogress transfers 

618 to complete and cancel all inprogress transfers. 

619 """ 

620 self._lock = threading.Lock() 

621 self._tracked_transfer_coordinators = set() 

622 

623 @property 

624 def tracked_transfer_coordinators(self): 

625 """The set of transfer coordinators being tracked""" 

626 with self._lock: 

627 # We return a copy because the set is mutable and if you were to 

628 # iterate over the set, it may be changing in length due to 

629 # additions and removals of transfer coordinators. 

630 return copy.copy(self._tracked_transfer_coordinators)

632 def add_transfer_coordinator(self, transfer_coordinator): 

633 """Adds a transfer coordinator of a transfer to be canceled if needed 

634 

635 :type transfer_coordinator: s3transfer.futures.TransferCoordinator 

636 :param transfer_coordinator: The transfer coordinator for the 

637 particular transfer

692 # of result(). We just want to ignore this and continue
