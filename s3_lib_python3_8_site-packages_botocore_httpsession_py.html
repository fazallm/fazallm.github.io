54def create_urllib3_context(ssl_version=None, cert_reqs=None, 

55 options=None, ciphers=None): 

56 """ This function is a vendored version of the same function in urllib3 

57 

58 We vendor this function to ensure that the SSL contexts we construct 

59 always use the std lib SSLContext instead of pyopenssl. 

60 """ 

61 context = SSLContext(ssl_version or ssl.PROTOCOL_SSLv23) 

62 

63 # Setting the default here, as we may have no ssl module on import 

64 cert_reqs = ssl.CERT_REQUIRED if cert_reqs is None else cert_reqs 

65 

66 if options is None: 

67 options = 0 

68 # SSLv2 is easily broken and is considered harmful and dangerous 

69 options |= OP_NO_SSLv2 

70 # SSLv3 has several problems and is now dangerous 

71 options |= OP_NO_SSLv3 

72 # Disable compression to prevent CRIME attacks for OpenSSL 1.0+ 

73 # (issue urllib3#309) 

74 options |= OP_NO_COMPRESSION 

75 

76 context.options |= options 

77 

78 if getattr(context, 'supports_set_ciphers', True): 

79 # Platform-specific: Python 2.6 

80 context.set_ciphers(ciphers or DEFAULT_CIPHERS) 

81 

82 context.verify_mode = cert_reqs 

83 if getattr(context, 'check_hostname', None) is not None: 

84 # Platform-specific: Python 3.2 

85 # We do our own verification, including fingerprints and alternative 

86 # hostnames. So disable it here 

87 context.check_hostname = False 

88 return context

91class ProxyConfiguration(object): 

92 """Represents a proxy configuration dictionary and additional settings. 

93 

94 This class represents a proxy configuration dictionary and provides utility 

95 functions to retreive well structured proxy urls and proxy headers from the 

96 proxy configuration dictionary. 

97 """ 

98 def __init__(self, proxies=None, proxies_settings=None): 

99 if proxies is None: 

100 proxies = {} 

101 if proxies_settings is None: 

102 proxies_settings = {} 

103 

104 self._proxies = proxies 

105 self._proxies_settings = proxies_settings 

106 

107 def proxy_url_for(self, url): 

108 """Retrieves the corresponding proxy url for a given url. """ 

109 parsed_url = urlparse(url) 

110 proxy = self._proxies.get(parsed_url.scheme) 

111 if proxy: 

112 proxy = self._fix_proxy_url(proxy) 

113 return proxy 

114 

115 def proxy_headers_for(self, proxy_url): 

116 """Retrieves the corresponding proxy headers for a given proxy url. """ 

117 headers = {} 

118 username, password = self._get_auth_from_url(proxy_url) 

119 if username and password: 

120 basic_auth = self._construct_basic_auth(username, password) 

121 headers['Proxy-Authorization'] = basic_auth 

122 return headers 

123 

124 @property 

125 def settings(self): 

126 return self._proxies_settings 

127 

128 def _fix_proxy_url(self, proxy_url): 

129 if proxy_url.startswith('http:') or proxy_url.startswith('https:'): 

130 return proxy_url 

131 elif proxy_url.startswith('//'): 

132 return 'http:' + proxy_url 

133 else: 

134 return 'http://' + proxy_url 

135 

136 def _construct_basic_auth(self, username, password): 

137 auth_str = '{0}:{1}'.format(username, password) 

138 encoded_str = b64encode(auth_str.encode('ascii')).strip().decode() 

139 return 'Basic {0}'.format(encoded_str) 

140 

141 def _get_auth_from_url(self, url): 

142 parsed_url = urlparse(url) 

143 try: 

144 return unquote(parsed_url.username), unquote(parsed_url.password) 

145 except (AttributeError, TypeError): 

146 return None, None

149class URLLib3Session(object): 

150 """A basic HTTP client that supports connection pooling and proxies. 

151 

152 This class is inspired by requests.adapters.HTTPAdapter, but has been 

153 boiled down to meet the use cases needed by botocore. For the most part 

154 this classes matches the functionality of HTTPAdapter in requests v2.7.0 

155 (the same as our vendored version). The only major difference of note is 

156 that we currently do not support sending chunked requests. While requests 

157 v2.7.0 implemented this themselves, later version urllib3 support this 

158 directly via a flag to urlopen so enabling it if needed should be trivial. 

159 """

162 proxies=None, 

163 timeout=None,

165 socket_options=None, 

166 client_cert=None, 

167 proxies_config=None,

171 proxies_settings=proxies_config)

173 'http': botocore.awsrequest.AWSHTTPConnectionPool, 

174 'https': botocore.awsrequest.AWSHTTPSConnectionPool, 

175 } 

176 if timeout is None: 

177 timeout = DEFAULT_TIMEOUT 

178 if not isinstance(timeout, (int, float)): 

179 timeout = Timeout(connect=timeout[0], read=timeout[1]) 

180 

181 self._cert_file = None 

182 self._key_file = None 

183 if isinstance(client_cert, str): 

184 self._cert_file = client_cert 

185 elif isinstance(client_cert, tuple): 

186 self._cert_file, self._key_file = client_cert 

187 

188 self._timeout = timeout 

189 self._max_pool_connections = max_pool_connections 

190 self._socket_options = socket_options

192 self._socket_options = []

253 def _setup_proxy_ssl_context(self, proxies_settings): 

254 proxy_ca_bundle = proxies_settings.get('proxy_ca_bundle') 

255 proxy_cert = proxies_settings.get('proxy_client_cert') 

256 if proxy_ca_bundle is None and proxy_cert is None: 

257 return None 

258 

259 context = self._get_ssl_context() 

260 try: 

261 # urllib3 disables this by default but we need 

262 # it for proper proxy tls negotiation. 

263 context.check_hostname = True

269 elif isinstance(proxy_cert, str):
