25#: A default dictionary that maps the logical names for session variables 

26#: to the specific environment variables and configuration file names 

27#: that contain the values for these variables. 

28#: When creating a new Session object, you can pass in your own dictionary 

29#: to remap the logical names or to add new logical names. You can then 

30#: get the current value for these variables by using the 

31#: ``get_config_variable`` method of the :class:`botocore.session.Session` 

32#: class. 

33#: These form the keys of the dictionary. The values in the dictionary 

34#: are tuples of (<config_name>, <environment variable>, <default value>, 

35#: <conversion func>). 

36#: The conversion func is a function that takes the configuration value 

37#: as an argument and returns the converted value. If this value is 

38#: None, then the configuration value is returned unmodified. This 

39#: conversion function can be used to type convert config values to 

40#: values other than the default values of strings. 

41#: The ``profile`` and ``config_file`` variables should always have a 

42#: None value for the first entry in the tuple because it doesn't make 

43#: sense to look inside the config file for the location of the config 

44#: file or for the default profile to use. 

45#: The ``config_name`` is the name to look for in the configuration file, 

46#: the ``env var`` is the OS environment variable (``os.environ``) to 

47#: use, and ``default_value`` is the value to use if no value is otherwise 

48#: found.

168class ConfigChainFactory(object): 

169 """Factory class to create our most common configuration chain case. 

170 

171 This is a convenience class to construct configuration chains that follow 

172 our most common pattern. This is to prevent ordering them incorrectly, 

173 and to make the config chain construction more readable. 

174 """ 

175 def __init__(self, session, environ=None): 

176 """Initialize a ConfigChainFactory. 

177 

178 :type session: :class:`botocore.session.Session` 

179 :param session: This is the session that should be used to look up 

180 values from the config file. 

181 

182 :type environ: dict 

183 :param environ: A mapping to use for environment variables. If this 

184 is not provided it will default to use os.environ. 

185 """ 

186 self._session = session 

187 if environ is None: 

188 environ = os.environ 

189 self._environ = environ 

190 

191 def create_config_chain(self, instance_name=None, env_var_names=None, 

192 config_property_names=None, default=None, 

193 conversion_func=None): 

194 """Build a config chain following the standard botocore pattern. 

195 

196 In botocore most of our config chains follow the the precendence: 

197 session_instance_variables, environment, config_file, default_value. 

198 

199 This is a convenience function for creating a chain that follow 

200 that precendence. 

201 

202 :type instance_name: str 

203 :param instance_name: This indicates what session instance variable 

204 corresponds to this config value. If it is None it will not be 

205 added to the chain. 

206 

207 :type env_var_names: str or list of str or None 

208 :param env_var_names: One or more environment variable names to 

209 search for this value. They are searched in order. If it is None 

210 it will not be added to the chain. 

211 

212 :type config_property_names: str/tuple or list of str/tuple or None 

213 :param config_property_names: One of more strings or tuples 

214 representing the name of the key in the config file for this 

215 config option. They are searched in order. If it is None it will 

216 not be added to the chain. 

217 

218 :type default: Any 

219 :param default: Any constant value to be returned. 

220 

221 :type conversion_func: None or callable 

222 :param conversion_func: If this value is None then it has no effect on 

223 the return type. Otherwise, it is treated as a function that will 

224 conversion_func our provided type. 

225 

226 :rvalue: ConfigChain 

227 :returns: A ConfigChain that resolves in the order env_var_names -> 

228 config_property_name -> default. Any values that were none are 

229 omitted form the chain. 

230 """ 

231 providers = [] 

232 if instance_name is not None: 

233 providers.append( 

234 InstanceVarProvider( 

235 instance_var=instance_name, 

236 session=self._session 

237 ) 

238 ) 

239 if env_var_names is not None: 

240 providers.extend(self._get_env_providers(env_var_names)) 

241 if config_property_names is not None: 

242 providers.extend( 

243 self._get_scoped_config_providers(config_property_names) 

244 ) 

245 if default is not None: 

246 providers.append(ConstantProvider(value=default)) 

247 

248 return ChainProvider( 

249 providers=providers, 

250 conversion_func=conversion_func, 

251 ) 

252 

253 def _get_env_providers(self, env_var_names): 

254 env_var_providers = [] 

255 if not isinstance(env_var_names, list): 

256 env_var_names = [env_var_names] 

257 for env_var_name in env_var_names: 

258 env_var_providers.append( 

259 EnvironmentProvider(name=env_var_name, env=self._environ) 

260 ) 

261 return env_var_providers 

262 

263 def _get_scoped_config_providers(self, config_property_names): 

264 scoped_config_providers = [] 

265 if not isinstance(config_property_names, list): 

266 config_property_names = [config_property_names] 

267 for config_property_name in config_property_names: 

268 scoped_config_providers.append( 

269 ScopedConfigProvider( 

270 config_var_name=config_property_name, 

271 session=self._session, 

272 ) 

273 ) 

274 return scoped_config_providers

277class ConfigValueStore(object): 

278 """The ConfigValueStore object stores configuration values.""" 

279 def __init__(self, mapping=None): 

280 """Initialize a ConfigValueStore. 

281 

282 :type mapping: dict 

283 :param mapping: The mapping parameter is a map of string to a subclass 

284 of BaseProvider. When a config variable is asked for via the 

285 get_config_variable method, the corresponding provider will be 

286 invoked to load the value. 

287 """ 

288 self._overrides = {} 

289 self._mapping = {} 

290 if mapping is not None: 

291 for logical_name, provider in mapping.items(): 

292 self.set_config_provider(logical_name, provider) 

293 

294 def get_config_variable(self, logical_name): 

295 """ 

296 Retrieve the value associeated with the specified logical_name 

297 from the corresponding provider. If no value is found None will 

298 be returned. 

299 

300 :type logical_name: str 

301 :param logical_name: The logical name of the session variable 

302 you want to retrieve. This name will be mapped to the 

303 appropriate environment variable name for this session as 

304 well as the appropriate config file entry. 

305 

306 :returns: value of variable or None if not defined. 

307 """ 

308 if logical_name in self._overrides: 

309 return self._overrides[logical_name] 

310 if logical_name not in self._mapping: 

311 return None 

312 provider = self._mapping[logical_name] 

313 return provider.provide() 

314 

315 def set_config_variable(self, logical_name, value): 

316 """Set a configuration variable to a specific value. 

317 

318 By using this method, you can override the normal lookup 

319 process used in ``get_config_variable`` by explicitly setting 

320 a value. Subsequent calls to ``get_config_variable`` will 

321 use the ``value``. This gives you per-session specific 

322 configuration values. 

323 

324 :: 

325 >>> # Assume logical name 'foo' maps to env var 'FOO' 

326 >>> os.environ['FOO'] = 'myvalue' 

327 >>> s.get_config_variable('foo') 

328 'myvalue' 

329 >>> s.set_config_variable('foo', 'othervalue') 

330 >>> s.get_config_variable('foo') 

331 'othervalue' 

332 

333 :type logical_name: str 

334 :param logical_name: The logical name of the session variable 

335 you want to set. These are the keys in ``SESSION_VARIABLES``. 

336 

337 :param value: The value to associate with the config variable. 

338 """ 

339 self._overrides[logical_name] = value 

340 

341 def clear_config_variable(self, logical_name): 

342 """Remove an override config variable from the session. 

343 

344 :type logical_name: str 

345 :param logical_name: The name of the parameter to clear the override 

346 value from. 

347 """ 

348 self._overrides.pop(logical_name, None) 

349 

350 def set_config_provider(self, logical_name, provider): 

351 """Set the provider for a config value. 

352 

353 This provides control over how a particular configuration value is 

354 loaded. This replaces the provider for ``logical_name`` with the new 

355 ``provider``. 

356 

357 :type logical_name: str 

358 :param logical_name: The name of the config value to change the config 

359 provider for. 

360 

361 :type provider: :class:`botocore.configprovider.BaseProvider` 

362 :param provider: The new provider that should be responsible for 

363 providing a value for the config named ``logical_name``. 

364 """ 

365 self._mapping[logical_name] = provider

379class ChainProvider(BaseProvider): 

380 """This provider wraps one or more other providers. 

381 

382 Each provider in the chain is called, the first one returning a non-None 

383 value is then returned. 

384 """ 

385 def __init__(self, providers=None, conversion_func=None): 

386 """Initalize a ChainProvider. 

387 

388 :type providers: list 

389 :param providers: The initial list of providers to check for values 

390 when invoked. 

391 

392 :type conversion_func: None or callable 

393 :param conversion_func: If this value is None then it has no affect on 

394 the return type. Otherwise, it is treated as a function that will 

395 transform provided value. 

396 """ 

397 if providers is None: 

398 providers = [] 

399 self._providers = providers 

400 self._conversion_func = conversion_func 

401 

402 def provide(self): 

403 """Provide the value from the first provider to return non-None. 

404 

405 Each provider in the chain has its provide method called. The first 

406 one in the chain to return a non-None value is the returned from the 

407 ChainProvider. When no non-None value is found, None is returned. 

408 """ 

409 for provider in self._providers: 

410 value = provider.provide() 

411 if value is not None: 

412 return self._convert_type(value) 

413 return None 

414 

415 def _convert_type(self, value): 

416 if self._conversion_func is not None: 

417 return self._conversion_func(value) 

418 return value 

419 

420 def __repr__(self): 

421 return '[%s]' % ', '.join([str(p) for p in self._providers])

452class ScopedConfigProvider(BaseProvider): 

453 def __init__(self, config_var_name, session): 

454 """Initialize ScopedConfigProvider. 

455 

456 :type config_var_name: str or tuple 

457 :param config_var_name: The name of the config variable to load from 

458 the configuration file. If the value is a tuple, it must only 

459 consist of two items, where the first item represents the section 

460 and the second item represents the config var name in the section. 

461 

462 :type session: :class:`botocore.session.Session` 

463 :param session: The botocore session to get the loaded configuration 

464 file variables from. 

465 """ 

466 self._config_var_name = config_var_name 

467 self._session = session 

468 

469 def provide(self): 

470 """Provide a value from a config file property.""" 

471 scoped_config = self._session.get_scoped_config() 

472 if isinstance(self._config_var_name, tuple): 

473 section_config = scoped_config.get(self._config_var_name[0]) 

474 if not isinstance(section_config, dict): 

475 return None 

476 return section_config.get(self._config_var_name[1]) 

477 return scoped_config.get(self._config_var_name) 

478 

479 def __repr__(self): 

480 return 'ScopedConfigProvider(config_var_name=%s, session=%s)' % ( 

481 self._config_var_name, 

482 self._session,

510class SectionConfigProvider(BaseProvider): 

511 """Provides a dictionary from a section in the scoped config 

512 

513 This is useful for retrieving scoped config variables (i.e. s3) that have 

514 their own set of config variables and resolving logic. 

515 """ 

516 def __init__(self, section_name, session, override_providers=None): 

517 self._section_name = section_name 

518 self._session = session 

519 self._scoped_config_provider = ScopedConfigProvider( 

520 self._section_name, self._session) 

521 self._override_providers = override_providers 

522 if self._override_providers is None: 

523 self._override_providers = {} 

524 

525 def provide(self): 

526 section_config = self._scoped_config_provider.provide() 

527 if section_config and not isinstance(section_config, dict): 

528 logger.debug("The %s config key is not a dictionary type, " 

529 "ignoring its value of: %s", self._section_name, 

530 section_config) 

531 return None 

532 for section_config_var, provider in self._override_providers.items(): 

533 provider_val = provider.provide() 

534 if provider_val is not None: 

535 if section_config is None: 

536 section_config = {} 

537 section_config[section_config_var] = provider_val 

538 return section_config 

539 

540 def __repr__(self): 

541 return ( 

542 'SectionConfigProvider(section_name=%s, ' 

543 'session=%s, override_providers=%s)' % ( 

544 self._section_name, self._session, 

545 self._override_providers, 

546 ) 

547 )
